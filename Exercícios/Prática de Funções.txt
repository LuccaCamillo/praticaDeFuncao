void main() {
  
  double n1 = 2;
  double n2 = 2;
  
  String operacao = "";
  
  switch (operacao){
      
    case "+":
    print(n1+n2);
    break;
      
    case "-":
    print(n1-n2);
    break;
    
    case "*":
    print(n1*n2);
    break;
      
    case "/":
    print(n1/n2);
    break;
      
    default:
    print("Escolha uma operação!");
    break;
      
  }
  
}
